const { pool } = require('./config/db');

async function diagnoseAuth() {
  try {
    console.log('==== DIAGNOZA PROBLEMU LOGOWANIA ====');
    console.log('Sprawdzam zmienne ≈õrodowiskowe:');
    console.log('JWT_SECRET ustawiony?', !!process.env.JWT_SECRET);
    if (!process.env.JWT_SECRET) {
      console.log('‚ö†Ô∏è UWAGA: Brak JWT_SECRET w zmiennych ≈õrodowiskowych!');
      console.log('Tworzƒô tymczasowy plik .env z JWT_SECRET');
      
      const fs = require('fs');
      fs.writeFileSync('.env', 'JWT_SECRET=amicus_secret_key_123\nJWT_EXPIRES_IN=7d\n');
      require('dotenv').config(); // Prze≈Çaduj zmienne ≈õrodowiskowe
      console.log('JWT_SECRET ustawiony teraz?', !!process.env.JWT_SECRET);
    }
    
    // Sprawd≈∫ po≈ÇƒÖczenie z bazƒÖ danych
    console.log('\nSprawdzam po≈ÇƒÖczenie z bazƒÖ danych...');
    try {
      await pool.query('SELECT 1');
      console.log('‚úÖ Po≈ÇƒÖczenie z bazƒÖ danych dzia≈Ça');
    } catch (dbError) {
      console.log('‚ùå Problem z po≈ÇƒÖczeniem z bazƒÖ danych:', dbError.message);
      return;
    }
    
    // Sprawd≈∫ dane u≈ºytkownika
    const email = 'admin@example.com';
    const password = 'Admin123';
    console.log(`\nSprawdzam u≈ºytkownika ${email} w bazie danych...`);
    
    // Pobierz u≈ºytkownika
    const userResult = await pool.query('SELECT * FROM users WHERE email = $1', [email]);
    
    if (userResult.rows.length === 0) {
      console.log(`‚ùå U≈ºytkownik ${email} nie istnieje w bazie danych!`);
      return;
    }
    
    const user = userResult.rows[0];
    console.log('‚úÖ U≈ºytkownik znaleziony w bazie danych');
    console.log('ID u≈ºytkownika:', user.id);
    console.log('Has≈Ço przechowywane w bazie:', user.password ? user.password.substring(0, 10) + '...' : 'NULL');
    
    // Sprawd≈∫ has≈Ço
    console.log('\nSprawdzam has≈Ço...');
    try {
      const isPasswordValid = await bcrypt.compare(password, user.password);
      console.log(`Weryfikacja has≈Ça: ${isPasswordValid ? '‚úÖ POPRAWNE' : '‚ùå NIEPOPRAWNE'}`);
      
      if (!isPasswordValid) {
        // Resetuj has≈Ço jako ostateczne rozwiƒÖzanie
        console.log('\nResetujƒô has≈Ço u≈ºytkownika...');
        const salt = await bcrypt.genSalt(10);
        const hashedPassword = await bcrypt.hash(password, salt);
        
        await pool.query(
          'UPDATE users SET password = $1, updated_at = $2 WHERE id = $3 RETURNING id',
          [hashedPassword, new Date(), user.id]
        );
        
        console.log('‚úÖ Has≈Ço zosta≈Ço zresetowane');
        console.log('Nowy hash has≈Ça:', hashedPassword);
      }
    } catch (bcryptError) {
      console.log('‚ùå B≈ÇƒÖd weryfikacji has≈Ça:', bcryptError.message);
    }
    
    // Spr√≥buj wygenerowaƒá token JWT
    console.log('\nGenerujƒô token JWT...');
    try {
      const token = jwt.sign(
        { id: user.id },
        process.env.JWT_SECRET || 'amicus_secret_key_123',
        { expiresIn: process.env.JWT_EXPIRES_IN || '7d' }
      );
      
      console.log('‚úÖ Token JWT wygenerowany poprawnie');
      console.log('Token:', token.substring(0, 20) + '...');
    } catch (jwtError) {
      console.log('‚ùå B≈ÇƒÖd generowania tokena JWT:', jwtError.message);
    }
    
    console.log('\n==== SUGESTIE NAPRAWY ====');
    console.log('1. Upewnij siƒô, ≈ºe plik .env zawiera JWT_SECRET');
    console.log('2. Sprawd≈∫ czy kontroler auth poprawnie obs≈Çuguje b≈Çƒôdy');
    console.log('3. Zrestartuj serwer po zmianach: systemctl restart amicus-backend.service');
    
  } catch (error) {
    console.error('B≈ÇƒÖd podczas diagnostyki:', error);
  } finally {
    pool.end();
  }
}

diagnoseAuth();

  host: process.env.DB_HOST,
  port: process.env.DB_PORT,
  user: process.env.DB_USER,
  password: process.env.DB_PASSWORD,
  database: process.env.DB_NAME,

    rejectUnauthorized: false // Akceptuje certyfikaty self-signed
  } : false

const bcrypt = require('bcryptjs');
const { pool } = require('./config/db');

async function createDefaultUser() {
  try {
    // Sprawdzamy czy u≈ºytkownik ju≈º istnieje
    const checkResult = await pool.query(
      'SELECT * FROM users WHERE email = $1',
      ['admin@example.com']
    );

    if (checkResult.rows.length > 0) {
      console.log('U≈ºytkownik admin@example.com ju≈º istnieje');
      return;
    }

    // Tworzymy hash has≈Ça
    const salt = await bcrypt.genSalt(10);
    const hashedPassword = await bcrypt.hash('Admin123', salt);

    // Dodajemy u≈ºytkownika
    const result = await pool.query(
      `INSERT INTO users (
        email, password, first_name, last_name, 
        status, created_at, updated_at
      ) VALUES ($1, $2, $3, $4, $5, $6, $7) RETURNING *`,
      [
        'admin@example.com',
        hashedPassword,
        'Admin',
        'User',
        'active',
        new Date(),
        new Date()
      ]
    );

    console.log('Utworzono u≈ºytkownika:', result.rows[0]);
  } catch (error) {
    console.error('B≈ÇƒÖd podczas tworzenia u≈ºytkownika:', error);
  } finally {
    // Zamknij po≈ÇƒÖczenie z bazƒÖ danych
    pool.end();
  }
}

async function debugAuth() {

    // 1. Sprawd≈∫, czy u≈ºytkownik istnieje w bazie
    const userQuery = await pool.query('SELECT * FROM users WHERE email = $1', [email]);
    
    if (userQuery.rows.length === 0) {
      console.log('‚ùå B≈ÅƒÑD: U≈ºytkownik nie istnieje w bazie danych');
      return;
    }
    
    const user = userQuery.rows[0];
    console.log('‚úÖ U≈ºytkownik znaleziony w bazie danych:', {
      id: user.id,
      email: user.email,
      password_length: user.password ? user.password.length : 0,
      password_hash: user.password ? user.password.substring(0, 20) + '...' : 'NULL'
    });
    
    // 2. Sprawd≈∫ has≈Ço
    try {
      const isPasswordValid = await bcrypt.compare(password, user.password);
      console.log(`Wynik weryfikacji has≈Ça: ${isPasswordValid ? '‚úÖ POPRAWNE' : '‚ùå NIEPOPRAWNE'}`);
      
      if (!isPasswordValid) {
        // Sprawd≈∫, czy has≈Ço nie jest jakim≈õ innym has≈Çem

        for (const testPwd of testPasswords) {
          try {
            const isOtherPassword = await bcrypt.compare(testPwd, user.password);
            if (isOtherPassword) {
              console.log(`üîç Znaleziono pasujƒÖce has≈Ço: "${testPwd}"`);
              break;
            }
          } catch (err) {}
        }
      }
    } catch (bcryptError) {
      console.log('‚ùå B≈ÅƒÑD przy weryfikacji has≈Ça:', bcryptError.message);
    }
    
    // 3. Zresetuj has≈Ço z nowym salt
    const salt = await bcrypt.genSalt(10);
    const newPasswordHash = await bcrypt.hash(password, salt);
    
    console.log('Nowy hash has≈Ça:', newPasswordHash);
    
    // 4. Zapisz nowe has≈Ço
    const updateResult = await pool.query(
      'UPDATE users SET password = $1, updated_at = $2 WHERE id = $3 RETURNING id',
      [newPasswordHash, new Date(), user.id]
    );
    
    console.log(`‚úÖ Has≈Ço zosta≈Ço zaktualizowane dla u≈ºytkownika ID ${updateResult.rows[0].id}`);
    
    // 5. Przetestuj nowe has≈Ço
    const updatedUserQuery = await pool.query('SELECT password FROM users WHERE id = $1', [user.id]);
    const updatedPasswordHash = updatedUserQuery.rows[0].password;
    
    const verifyNewPassword = await bcrypt.compare(password, updatedPasswordHash);
    console.log(`Weryfikacja nowego has≈Ça: ${verifyNewPassword ? '‚úÖ POPRAWNE' : '‚ùå NIEPOPRAWNE'}`);

      console.log('3. Dodaj dodatkowe informacje w logach autoryzacji, aby zrozumieƒá dok≈Çadniej problem');
    } else {
      console.log('Mamy problem z has≈Çem - haszowanie nie dzia≈Ça poprawnie.');
    }
    
  } catch (error) {
    console.error('B≈ÅƒÑD DIAGNOSTYKI:', error);
  } finally {

  }

const jwt = require('jsonwebtoken');
const userRepository = require('../repositories/userRepository');

exports.verifyToken = async (req, res, next) => {
  try {
    // Pobierz token z nag≈Ç√≥wka Authorization
    const authHeader = req.headers.authorization;
    if (!authHeader || !authHeader.startsWith('Bearer ')) {
      return res.status(401).json({ message: 'Brak tokenu uwierzytelniajƒÖcego' });
    }

    const token = authHeader.split(' ')[1];

    // Weryfikuj token
    const decoded = jwt.verify(token, process.env.JWT_SECRET);

    // Zapisz ID u≈ºytkownika w obiekcie ≈ºƒÖdania
    req.userId = decoded.id;

    // Sprawd≈∫, czy u≈ºytkownik istnieje
    const user = await userRepository.findById(decoded.id);
    if (!user) {
      return res.status(401).json({ message: 'Nieprawid≈Çowy token' });
    }

    // Dodaj obiekt u≈ºytkownika do ≈ºƒÖdania (bez has≈Ça)
    const { password, ...userWithoutPassword } = user;
    req.user = userWithoutPassword;

    next();
  } catch (error) {
    if (error.name === 'JsonWebTokenError') {
      return res.status(401).json({ message: 'Nieprawid≈Çowy token' });
    }
    if (error.name === 'TokenExpiredError') {
      return res.status(401).json({ message: 'Token wygas≈Ç' });
    }
    return res.status(500).json({ message: 'B≈ÇƒÖd serwera' });
  }
};

===== middleware/errorHandler.js =====
const { ValidationError } = require('express-validator');
const winston = require('winston');

// Konfiguracja loggera
const logger = winston.createLogger({
  level: 'error',
  format: winston.format.combine(
    winston.format.timestamp(),
    winston.format.json()
  ),
  transports: [
    new winston.transports.File({ filename: 'error.log' }),
    new winston.transports.Console()
  ]
});

// Niestandardowa klasa b≈Çƒôdu aplikacji
class AppError extends Error {
  constructor(message, statusCode) {
    super(message);
    this.statusCode = statusCode;
    this.isOperational = true;
    Error.captureStackTrace(this, this.constructor);
  }
}

// Middleware obs≈Çugi b≈Çƒôd√≥w
const errorHandler = (err, req, res, next) => {
  // Domy≈õlny kod statusu i komunikat
  let statusCode = err.statusCode || 500;
  let message = err.message || 'WystƒÖpi≈Ç b≈ÇƒÖd serwera';

  // Logowanie b≈Çƒôdu
  logger.error({
    message: err.message,
    stack: err.stack,
    path: req.path,
    method: req.method,
    statusCode
  });

  // Sprawd≈∫, czy to b≈ÇƒÖd walidacji (express-validator)
  if (Array.isArray(err) && err.every(e => e instanceof ValidationError)) {
    statusCode = 400;
    message = err.map(e => ({ 
      field: e.param, 
      message: e.msg 
    }));
  }

  // B≈ÇƒÖd JWT
  if (err.name === 'JsonWebTokenError' || err.name === 'TokenExpiredError') {
    statusCode = 401;
    message = 'Nieprawid≈Çowy token uwierzytelniajƒÖcy';
  }

  // B≈Çƒôdy bazy danych
  if (err.code === '23505') {  // PostgreSQL unique violation
    statusCode = 409;
    message = 'Rekord ju≈º istnieje w bazie danych';
  }

  // B≈Çƒôdy operacyjne mogƒÖ ujawniaƒá szczeg√≥≈Çy
  // B≈Çƒôdy programistyczne powinny pokazywaƒá og√≥lny komunikat w produkcji
  const isProduction = process.env.NODE_ENV === 'production';
  
  if (!err.isOperational && isProduction) {
    message = 'WystƒÖpi≈Ç b≈ÇƒÖd serwera';
  }

  // W trybie deweloperskim zwracamy wiƒôcej informacji
  return res.status(statusCode).json({
    status: 'error',
    message,
    ...(process.env.NODE_ENV === 'development' && !err.isOperational && { 
      stack: err.stack 
    })
  });
};

// Middleware obs≈Çugi nieistniejƒÖcych tras
const notFoundHandler = (req, res, next) => {
  const error = new AppError(`Nie znaleziono - ${req.originalUrl}`, 404);
  next(error);
};

module.exports = {
  AppError,
  errorHandler,
  notFoundHandler
};

const { body, validationResult } = require('express-validator');

// Sprawdzenie wynik√≥w walidacji
const validateRequest = (req, res, next) => {
  const errors = validationResult(req);
  if (!errors.isEmpty()) {
    return res.status(400).json({ 
      status: 'error',
      errors: errors.array() 
    });
  }
  next();
};

const bcrypt = require('bcryptjs');
const jwt = require('jsonwebtoken');
const userRepository = require('../repositories/userRepository');
const organizationRepository = require('../repositories/organizationRepository');

class AuthService {
  async register(userData, organizationData = null) {
    // Sprawd≈∫, czy u≈ºytkownik o takim emailu ju≈º istnieje
    const existingUser = await userRepository.findByEmail(userData.email);
    if (existingUser) {
      throw new Error('U≈ºytkownik o takim adresie email ju≈º istnieje');
    }

    // Hashuj has≈Ço
    const salt = await bcrypt.genSalt(10);
    const hashedPassword = await bcrypt.hash(userData.password, salt);
    
    // ZastƒÖp has≈Ço jawne hashem
    userData.password = hashedPassword;

    // Utw√≥rz u≈ºytkownika
    const newUser = await userRepository.create(userData);

    let organizationResult = null;
    
    // Je≈õli przekazano dane organizacji, utw√≥rz jƒÖ i przypisz u≈ºytkownika jako administratora
    if (organizationData && organizationData.name) {
      const newOrganization = await organizationRepository.create(organizationData);
      await organizationRepository.addUserToOrganization(newOrganization.id, newUser.id, 'admin');
      organizationResult = newOrganization;
    }

    // Generuj token JWT
    const token = this.generateToken(newUser.id);

    return {
      user: newUser,
      organization: organizationResult,
      token
    };
  }

  async login(email, password) {
    console.log('[AUTH] Rozpoczƒôcie logowania dla:', email);
    
    try {
      // Znajd≈∫ u≈ºytkownika
      console.log('[AUTH] Szukanie u≈ºytkownika w bazie...');
      const user = await userRepository.findByEmail(email);
      
      if (!user) {
        console.log('[AUTH] ‚ùå U≈ºytkownik nie znaleziony');
        throw new Error('Nieprawid≈Çowe dane logowania');
      }
      
      console.log('[AUTH] ‚úÖ U≈ºytkownik znaleziony, ID:', user.id);

      // Sprawd≈∫ has≈Ço
      console.log('[AUTH] Weryfikacja has≈Ça...');
      const isPasswordValid = await bcrypt.compare(password, user.password);
      
      if (!isPasswordValid) {
        console.log('[AUTH] ‚ùå Nieprawid≈Çowe has≈Ço');
        throw new Error('Nieprawid≈Çowe dane logowania');
      }
      
      console.log('[AUTH] ‚úÖ Has≈Ço zweryfikowane poprawnie');

      // Pobierz organizacje u≈ºytkownika
      console.log('[AUTH] Pobieranie organizacji u≈ºytkownika...');
      let organizations = [];
      try {
        organizations = await organizationRepository.getUserOrganizations(user.id);
        console.log('[AUTH] ‚úÖ Pobrano organizacje:', organizations.length);
      } catch (orgError) {
        console.log('[AUTH] ‚ö†Ô∏è B≈ÇƒÖd podczas pobierania organizacji:', orgError.message);
        // Kontynuuj mimo b≈Çƒôdu organizacji
        organizations = [];
      }

      // Generuj token JWT
      console.log('[AUTH] Generowanie tokenu JWT...');
      const token = this.generateToken(user.id);
      console.log('[AUTH] ‚úÖ Token wygenerowany');

      // Usu≈Ñ has≈Ço z danych u≈ºytkownika przed zwr√≥ceniem
      const userToReturn = { ...user };
      delete userToReturn.password;
      
      console.log('[AUTH] üéâ Logowanie zako≈Ñczone sukcesem');
      return {
        user: userToReturn,
        organizations,
        token
      };
    } catch (error) {
      console.log('[AUTH] ‚ùå B≈ÇƒÖd podczas logowania:', error.message);
      throw error;
    }
  }

  generateToken(userId) {
    return jwt.sign(
      { id: userId },
      process.env.JWT_SECRET || 'amicus_default_secret',
      { expiresIn: process.env.JWT_EXPIRES_IN || '7d' }
    );
  }

      throw new Error('U≈ºytkownik nie znaleziony');
    }

    // Pobierz organizacje u≈ºytkownika

    // Usu≈Ñ has≈Ço z obiektu u≈ºytkownika
    const userToReturn = { ...user };
    delete userToReturn.password;
    
    return {
      user: userToReturn,
      organizations
    };
  }
